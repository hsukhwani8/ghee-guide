                    <h3 class="text-2xl font-bold mb-6 text-center" style="color: #43AA8B;">Bilona Ghee</h3>

                    <img src="https://i.postimg.cc/fWvRpv4t/Image-fx-1.png" alt="Traditional bilona ghee making process" class="rounded-lg shadow-md mb-6 w-full object-cover">

                    <div class="space-y-5">

                        <div class="flow-step" style="border-color: #43AA8B;"><h4 class="font-bold">1. Sacred Source (Ahimsa)</h4><p class="text-sm">Milk is taken from stress-free Desi A2 cows only after their calves have been fully fed.</p></div>

                        <div class="flow-step" style="border-color: #43AA8B;"><h4 class="font-bold">2. Whole Milk Curdling</h4><p class="text-sm">The entire volume of milk is gently simmered and cultured into probiotic-rich curd.</p></div>

                        <div class="flow-step" style="border-color: #43AA8B;"><h4 class="font-bold">3. The Bilona Churn</h4><p class="text-sm">This curd is slowly hand-churned in a wooden churner (a 'bilona') to separate pure white butter (*makkhan*).</p></div>

                        <div class="flow-step" style="border-color: #43AA8B;"><h4 class="font-bold">4. Slow Clarification</h4><p class="text-sm">The butter is heated on a low, wood-fueled fire to lovingly craft pure, aromatic, and spiritually potent ghee.</p></div>

                    </div>

                </div>

                        <div class="flow-step" style="border-color: #E53E3E;"><h4 class="font-bold">1. Mass Collection</h4><p class="text-sm">Milk is sourced from various farms, often from cross-bred A1 cows optimised for high yield.</p></div>

                        <div class="flow-step" style="border-color: #E53E3E;"><h4 class="font-bold">2. Cream Separation</h4><p class="text-sm">Industrial machines rapidly strip the cream from milk. The nutritious skimmed milk is used for other products.</p></div>

                        <div class="flow-step" style="border-color: #E53E3E;"><h4 class="font-bold">3. High-Temperature Processing</h4><p class="text-sm">The collected cream is boiled at extremely high temperatures for fast, high-volume ghee production.</p></div>

                        <div class="flow-step" style="border-color: #E53E3E;"><h4 class="font-bold">4. Diminished Result</h4><p class="text-sm">A substance called 'ghee' is produced, but its delicate vitamins and subtle life force (*prana*) are significantly damaged.</p></div>

        const wrapLabel = (label, maxLength = 16) => {

            if (typeof label !== 'string' || label.length <= maxLength) return label;

            const words = label.split(' ');

            const lines = [];

            let currentLine = '';

            for (const word of words) {

                if ((currentLine + ' ' + word).trim().length > maxLength && currentLine.length > 0) {

                    lines.push(currentLine);

                    currentLine = word;

                } else {

                    currentLine = (currentLine + ' ' + word).trim();

                }

            }

            if (currentLine.length > 0) {

                lines.push(currentLine);

            }

            return lines;

        };
